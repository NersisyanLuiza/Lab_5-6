#include <iostream>
#include <fstream>
#include <string>
#include <vector>
#include <filesystem>
#include <algorithm>

#ifdef _WIN32
#include <windows.h>
#endif

using namespace std;
namespace fs = std::filesystem;

// --- Encryption Engine (Գաղտնագրման շարժիչ) ---
class EncryptionEngine {
public:
    // XOR գաղտնագրման ֆունկցիա
    static void xor_process(vector<char>& data, const string& key) {
        for (size_t i = 0; i < data.size(); ++i) {
            data[i] ^= key[i % key.length()];
        }
    }

    static bool process_file(const fs::path& filePath, const string& key, bool encrypt) {
        // Ստուգում ենք արդյոք ֆայլը արդեն գաղտնագրված է կամ հակառակը
        if (encrypt && filePath.extension() == ".encrypted") return false;
        if (!encrypt && filePath.extension() != ".encrypted") return false;

        // 1. Կարդալ ֆայլը
        ifstream inFile(filePath, ios::binary);
        if (!inFile) return false;
        vector<char> data((istreambuf_iterator<char>(inFile)), istreambuf_iterator<char>());
        inFile.close();

        // 2. Գաղտնագրել/Վերծանել
        xor_process(data, key);

        // 3. Որոշել նոր ֆայլի անունը
        fs::path outPath = filePath;
        if (encrypt) {
            outPath += ".encrypted";
        }
        else {
            string p = outPath.string();
            outPath = p.substr(0, p.find_last_of('.'));
        }

        // 4. Պահպանել նոր ֆայլը
        ofstream outFile(outPath, ios::binary);
        if (!outFile) return false;
        outFile.write(data.data(), data.size());
        outFile.close();

        // 5. Ջնջել բնօրինակը (ըստ պահանջի)
        fs::remove(filePath);
        return true;
    }
};

// --- USB Drive Detection (USB կրիչների հայտնաբերում) ---
void scanDrives() {
    cout << "Հայտնաբերված USB կրիչներ:" << endl;
#ifdef _WIN32
    char driveBuffer[256];
    GetLogicalDriveStringsA(sizeof(driveBuffer), driveBuffer);
    char* drive = driveBuffer;
    int count = 0;
    while (*drive) {
        if (GetDriveTypeA(drive) == DRIVE_REMOVABLE) {
            cout << ++count << ". " << drive << endl;
        }
        drive += strlen(drive) + 1;
    }
    if (count == 0) cout << "Ոչ մի USB կրիչ չի հայտնաբերվել:" << endl;
#else
    cout << "Այս պլատֆորմի համար սկանավորումը դեռ հասանելի չէ (TODO):" << endl;
#endif
}

// --- File Operations (Ֆայլային գործողություններ) ---
void listFiles(const string& path) {
    if (!fs::exists(path)) {
        cout << "Ուղին գոյություն չունի:" << endl;
        return;
    }
    cout << "Ֆայլերի ցուցակ (" << path << "):" << endl;
    for (const auto& entry : fs::directory_iterator(path)) {
        cout << (entry.is_directory() ? "[DIR]  " : "[FILE] ") << entry.path().filename().string() << endl;
    }
}

// --- Հիմնական ինտերֆեյս ---
int main(int argc, char* argv[]) {
    SetConsoleOutputCP(65001);
    SetConsoleCP(65001);
    setlocale(LC_ALL, "Armenian");

    if (argc < 2) {
        cout << "Օգտագործում:" << endl;
        cout << "  protectusb scan" << endl;
        cout << "  protectusb list [ուղի]" << endl;
        cout << "  protectusb protect [ուղի]" << endl;
        cout << "  protectusb unprotect [ուղի]" << endl;
        cout << "  protectusb encrypt [ֆայլ]" << endl;
        cout << "  protectusb decrypt [ֆայլ]" << endl;
        return 0;
    }

    string command = argv[1];

    if (command == "scan") {
        scanDrives();
    }
    else if (command == "list" && argc > 2) {
        listFiles(argv[2]);
    }
    else if (command == "encrypt" || command == "decrypt") {
        if (argc < 3) {
            cout << "Նշեք ֆայլի ուղին:" << endl;
            return 1;
        }
        string key;
        cout << "Մուտքագրեք բանալին: ";
        cin >> key;
        if (EncryptionEngine::process_file(argv[2], key, (command == "encrypt"))) {
            cout << "Գործողությունը հաջողվեց:" << endl;
        }
        else {
            cout << "Սխալ ֆայլի մշակման ժամանակ:" << endl;
        }
    }
    else if (command == "protect" || command == "unprotect") {
        if (argc < 3) {
            cout << "Նշեք ուղին:" << endl;
            return 1;
        }
        string key;
        cout << "Մուտքագրեք գաղտնագրման բանալի: ";
        cin >> key;

        string targetDir = argv[2];
        bool isEncrypt = (command == "protect");
        int count = 0;

        cout << (isEncrypt ? "Պաշտպանում եմ..." : "Վերականգնում եմ...") << endl;
        cout << "────────────────────────────────" << endl;

        try {
            for (const auto& entry : fs::recursive_directory_iterator(targetDir)) {
                if (entry.is_regular_file()) {
                    if (EncryptionEngine::process_file(entry.path(), key, isEncrypt)) {
                        cout << "Մշակվեց: " << entry.path().filename() << endl;
                        count++;
                    }
                }
            }
            cout << "────────────────────────────────" << endl;
            cout << "Գործողությունն ավարտվեց: Մշակվեց " << count << " ֆայլ:" << endl;
        }
        catch (const exception& e) {
            cout << "Սխալ: " << e.what() << endl;
        }
    }
    else {
        cout << "Անհայտ հրաման:" << endl;
    }

    return 0;
}
